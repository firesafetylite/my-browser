function navigate(input) { errorMsg.style.display = 'none'; const url = normalizeURL(input); if (!url) return; urlBar.value = url; // Check if Ultraviolet is available if (typeof __uv$config === 'undefined') { errorMsg.textContent = 'Ultraviolet not loaded. Make sure uv/uv.bundle.js and uv/uv.config.js are present.'; errorMsg.style.display = 'block'; setStatus('Error: UV not loaded.'); return; } const proxied = __uv$config.prefix + __uv$config.encodeUrl(url); homeScreen.style.display = 'none'; frame.style.display = 'block'; frame.src = proxied; setStatus('Loading: ' + url); frame.onload = () => setStatus(url); } // Toolbar buttons document.getElementById('back-btn').onclick = () => { try { frame.contentWindow.history.back(); } catch(e){} }; document.getElementById('fwd-btn').onclick = () => { try { frame.contentWindow.history.forward(); } catch(e){} }; document.getElementById('reload-btn').onclick = () => { try { frame.contentWindow.location.reload(); } catch(e) { frame.src = frame.src; } }; document.getElementById('home-btn').onclick = () => { frame.style.display = 'none'; frame.src = ''; urlBar.value = ''; homeScreen.style.display = 'flex'; setStatus(''); }; document.getElementById('go-btn').onclick = () => navigate(urlBar.value); urlBar.addEventListener('keydown', e => { if (e.key === 'Enter') navigate(urlBar.value); }); document.getElementById('home-go').onclick = () => navigate(document.getElementById('home-input').value); document.getElementById('home-input').addEventListener('keydown', e => { if (e.key === 'Enter') navigate(document.getElementById('home-input').value); });
